# bibly

syntax: ./bibly "cite1" "cite2" ... "citeN"

cite can be:

single verse: jn3:16
chapter span: jn1-3
verse span:   jn3:16-20
verse hop:    jn3:16,18,20

any combination of the 
above separated by commas

# bfind

syntax: ./bfind "search1" "search2" ... "search3"

searches for text

# bpick

syntax: ./bpick

displays a random bible verse
